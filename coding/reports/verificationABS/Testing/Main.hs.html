   55 -- | Testing behaviour of susceptible agent
   56 --    a susceptible agent makes on average contact
   57 --    with contact rate other agents per time-unit
   58 --    => when running N susceptible agents for 1.0
   59 --    time-unit then
   60 --    N * infectivity * contactRate * (infectedAgentsCount / totalAgentsCount) 
   61 --    agents should become infected
   62 --   NOTE: this also allows to estimate the dt: to 
   63 --   achieve a sufficiently close match one selects
   64 --   a very small epsilon and then reduces the dt
   65 --   until the average falls into the epsilon environment
   66 -- NOTE: this is black-box verification
   67 testCaseSusceptible :: RandomGen g
   68                     => g 
   69                     -> [SIRState]
   70                     -> Bool
   71 testCaseSusceptible g0 otherAgents = diff <= eps
   72   where
   73     n    = 10000 -- TODO: how to select a 'correct' number of runs?
   74     eps  = 0.1   -- TODO: how to select a 'correct' epsilon? probably it depends on the ratio between dt and contact rate?
   75     dt   = 0.1   -- TODO: how to select a 'correct' dt? when the other parameters (n and eps) are selected 'correctly' then adjust dt until the test succeeds
   76     diff = testSusceptible g0 n dt
   77 
   78     testSusceptible :: RandomGen g
   79                     => g      -- ^ initial RNG
   80                     -> Int    -- ^ number of runs
   81                     -> DTime  -- ^ time-delta to use
   82                     -> Double -- ^ returns difference to target
   83     testSusceptible g0 n dt 
   84         = abs (target - countFract)
   85       where
   86         -- we have 3 other agents, each in one of the states
   87         -- this means, that this susceptible agent will pick
   88         -- on average an Infected with a probability of 1/3
   89         otherAgentsCount  = length otherAgents
   90         infOtherAgents    = length $ filter (Infected==) otherAgents
   91         infToNonInfRatio
   92           -- prevent division by zero
   93           = if 0 == otherAgentsCount 
   94               then 0
   95               else fromIntegral infOtherAgents / fromIntegral otherAgentsCount
   96 
   97         count      = testSusceptibleAux otherAgents g0 n 0
   98         countFract = fromIntegral count / fromIntegral n
   99         -- multiply with contactRate because we make on 
  100         -- average contact with contactRate other agents 
  101         -- per time-unit (we are running the agent for
  102         -- 1.0 time-unit)
  103         -- also multiply with ratio of infected to non-infected
  104         --   TODO: can we extract the formula of the SD approach
  105         --   here? should be possible, then we can prove that our
  106         --   ABS approach is indeed a valid SD approximation
  107         --   (due to averaging) 
  108         target = infectivity * contactRate * infToNonInfRatio

  110         testSusceptibleAux :: RandomGen g 
  111                           => [SIRState]
  112                           -> g
  113                           -> Int
  114                           -> Int
  115                           -> Int
  116         testSusceptibleAux _ _ 0 count = count
  117         testSusceptibleAux otherAgents g n count 
  118             = testSusceptibleAux otherAgents g'' (n-1) count'
  119           where
  120             (g', g'')   = split g
  121 
  122             stepsCount  = floor (1.0 / dt)
  123             steps       = replicate stepsCount (dt, Nothing)
  124 
  125             ret         = embed (testSusceptibleSF otherAgents g') ((), steps)
  126             gotInfected = True `elem` ret
  127 
  128             count'      = if gotInfected then count + 1 else count
  129 
  130         testSusceptibleSF :: RandomGen g 
  131                           => [SIRState]
  132                           -> g
  133                           -> SF () Bool
  134         testSusceptibleSF otherAgents g = proc _ -> do
  135           ret <- susceptibleAgent g -< otherAgents
  136           case ret of 
  137             Susceptible -> returnA -< False
  138             Infected    -> returnA -< True
  139             Recovered   -> returnA -< False -- TODO: should never occur, can we test this? seems not so, but we can pretty easily guarantee it due to simplicity of code
  140 
  141 -- | Testing behaviour of infected agent
  142 --   run test until agent recovers, which happens
  143 --   on average after illnessDuration
  144 --   we are running this test a larger number of 
  145 --   times N and averaging the durations of all
  146 --   agents until their recovery
  147 --   should be within an epsilon of illnessDuration
  148 -- NOTE: this is black-box verification
  149 testCaseInfected :: RandomGen g 
  150                  => g
  151                  -> Bool
  152 testCaseInfected g0 = diff <= eps
  153   where
  154     n    = 10000 -- TODO: how to select a 'correct' number of runs?
  155     eps  = 0.5   -- TODO: how to select a 'correct' epsilon? probably it depends on ratio between dt and illnessduration?
  156     dt   = 0.1   -- TODO: how to select a 'correct' dt? when the other parameters (n and eps) are selected 'correctly' then adjust dt until the test succeeds
  157     diff = testInfected g0 n dt

  159     testInfected :: RandomGen g 
  160                 => g      -- ^ initial RNG
  161                 -> Int    -- ^ number of runs
  162                 -> DTime  -- ^ time-delta to use
  163                 -> Double   -- ^ difference to target
  164     testInfected g0 n dt = abs (target - durationsAvg)
  165       where
  166         durations    = testInfectedAux g0 n []
  167         durationsAvg = sum durations / fromIntegral (length durations)
  168         
  169         target = illnessDuration
  170         
  171         testInfectedAux :: RandomGen g 
  172                         => g
  173                         -> Int
  174                         -> [Double]
  175                         -> [Double]
  176         testInfectedAux _ 0 acc = acc
  177         testInfectedAux g n acc 
  178             = testInfectedAux g'' (n-1) acc'
  179           where
  180             (g', g'')    = split g
  181             steps        = repeat (dt, Nothing)
  182             evts         = embed (testInfectedSF g') ((), steps)
  183             -- there will always be an event, testInfectedSF will eventuall return an event 
  184             (Event t)    = fromJust $ find isEvent evts
  185             acc'         = t : acc
  186 
  187         testInfectedSF :: RandomGen g 
  188                       => g
  189                       -> SF () (Event Time)
  190         testInfectedSF g = proc _ -> do
  191           ret <- infectedAgent g -< []
  192           t <- time -< ()
  193           case ret of 
  194             Susceptible -> returnA -< NoEvent -- TODO: should never occur, can we test this? seems not so, but we can pretty easily guarantee it due to simplicity of code
  195             Infected    -> returnA -< NoEvent
  196             Recovered   -> returnA -< Event t

  210 -- | Compare the dynamics of ABS and SD approach
  211 --   should be on average the same
  212 --   we are calculating a large number of ABS
  213 --   replications N and average them, this will
  214 --   then be compared to the SD dynamics (note
  215 --   that we only need to calculate the SD

  217 -- TODO: doesnt really work... comparing SD and ABS dynamics is

  220 testDynamics :: RandomGen g 
  221              => g
  222              -> IO ()
  223 testDynamics g = do
  224     let popSize = 100 :: Double
  225     let infCount = 1   :: Double
  226 
  227     let sdDyns  = runSD popSize infCount 150 0.1
  228     absDynss <- forM ([1..10] :: [Int]) (\_ -> do
  229       g' <- newStdGen
  230       return $ runABS g' (floor popSize) (floor infCount) 150 0.1)
  231 
  232     let absDynssFiltered = filter okDynamics absDynss
  233     let absDyns = averageAbsDynamics absDynssFiltered
  234 
  235     let (sdSus, sdInf, sdRec)    = unzip3 sdDyns
  236     let (absSus, absInf, absRec) = unzip3 absDyns
  237 
  238     let nmseSus = nmse sdSus absSus
  239     let nmseInf = nmse sdInf absInf
  240     let nmseRec = nmse sdRec absRec
  241 
  242     let sdfilename  = "sd_" ++ show popSize ++ ".m"
  243     let absfilename = "abs_" ++ show popSize ++ ".m"
  244 
  245     writeAggregatesToFile sdfilename sdDyns
  246     writeAggregatesToFile absfilename absDyns
  247 
  248     print (nmseSus, nmseInf, nmseRec)

  251     okDynamics :: [(Double, Double, Double)] -> Bool
  252     okDynamics = (>50) . third . last 
  253       where
  254         third (_, _, x) = x
  255 
  256     -- | Normalized Mean Square Error
  257     -- Assuming xs and ys are same length 
  258     nmse :: [Double]  -- ^ xs
  259          -> [Double]  -- ^ ys
  260          -> Double    -- ^ nmse 
  261     nmse xs ys = nmseSum / n
  262       where
  263         n = fromIntegral $ length xs
  264         xsNorm = sum xs / n
  265         ysNorm = sum ys / n
  266 
  267         nmseSum = sum $ zipWith (\x y -> ((x - y) ** 2) / (xsNorm * ysNorm)) xs ys
  268 
  269     -- | Averaging, but ignoring cases where the infected recover
  270     -- without really starting the dynamics
  271     averageAbsDynamics :: [[(Double, Double, Double)]]
  272                        -> [(Double, Double, Double)]
  273     averageAbsDynamics []         = []
  274     averageAbsDynamics (ds : dss) = dsAvgs
  275       where
  276         -- TODO filter out "invalid dynamics"
  277         n = 1 + fromIntegral (length dss)
  278         dsSums = sumAbsDyns dss ds 
  279         dsAvgs = map (\(ss, is, rs) -> (ss / n, is / n, rs / n)) dsSums
  280 
  281         sumAbsDyns :: [[(Double, Double, Double)]]
  282                    -> [(Double, Double, Double)]
  283                    -> [(Double, Double, Double)]
  284         sumAbsDyns []  acc        = acc
  285         sumAbsDyns (ds : dss) acc = sumAbsDyns dss acc'
  286           where
  287             acc' = zipWith tripleSum ds acc
  288 
  289             tripleSum :: (Double, Double, Double)
  290                       -> (Double, Double, Double)
  291                       -> (Double, Double, Double)
  292             tripleSum (x1, y1, z1) (x2, y2, z2) 
  293               = (x1 + x2, y1 + y2, z1 + z2)
