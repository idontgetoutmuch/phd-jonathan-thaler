   22 writeAggregatesToFile :: String -> [(Double, Double, Double)] -> IO ()
   23 writeAggregatesToFile fileName dynamics = do
   24   fileHdl <- openFile fileName WriteMode
   25   hPutStrLn fileHdl "dynamics = ["
   26   mapM_ (hPutStrLn fileHdl . sirAggregateToString) dynamics
   27   hPutStrLn fileHdl "];"
   28 
   29   hPutStrLn fileHdl "susceptible = dynamics (:, 1);"
   30   hPutStrLn fileHdl "infected = dynamics (:, 2);"
   31   hPutStrLn fileHdl "recovered = dynamics (:, 3);"
   32   hPutStrLn fileHdl "totalPopulation = susceptible(1) + infected(1) + recovered(1);"
   33 
   34   hPutStrLn fileHdl "susceptibleRatio = susceptible ./ totalPopulation;"
   35   hPutStrLn fileHdl "infectedRatio = infected ./ totalPopulation;"
   36   hPutStrLn fileHdl "recoveredRatio = recovered ./ totalPopulation;"
   37 
   38   hPutStrLn fileHdl "steps = length (susceptible);"
   39   hPutStrLn fileHdl "indices = 0 : steps - 1;"
   40 
   41   hPutStrLn fileHdl "figure"
   42   hPutStrLn fileHdl "plot (indices, susceptibleRatio.', 'color', 'blue', 'linewidth', 2);"
   43   hPutStrLn fileHdl "hold on"
   44   hPutStrLn fileHdl "plot (indices, infectedRatio.', 'color', 'red', 'linewidth', 2);"
   45   hPutStrLn fileHdl "hold on"
   46   hPutStrLn fileHdl "plot (indices, recoveredRatio.', 'color', 'green', 'linewidth', 2);"
   47 
   48   hPutStrLn fileHdl "set(gca,'YTick',0:0.05:1.0);"
   49   
   50   hPutStrLn fileHdl "xlabel ('Time');"
   51   hPutStrLn fileHdl "ylabel ('Population Ratio');"
   52   hPutStrLn fileHdl "legend('Susceptible','Infected', 'Recovered');"
   53 
   54   hClose fileHdl
   55 
   56 sirAggregateToString :: (Double, Double, Double) -> String
   57 sirAggregateToString (susceptibleCount, infectedCount, recoveredCount) =
   58   printf "%f" susceptibleCount
   59   ++ "," ++ printf "%f" infectedCount
   60   ++ "," ++ printf "%f" recoveredCount
   61   ++ ";"
