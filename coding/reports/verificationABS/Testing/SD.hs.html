1 {-# LANGUAGE Arrows #-}
    2 module SD 
    3   (
    4     runSD
    5   ) where
    6 
    7 import FRP.Yampa
    8 
    9 import SIR
   10 
   11 runSD :: Double
   12       -> Double
   13       -> Time
   14       -> DTime
   15       -> [(Double, Double, Double)]
   16 runSD populationSize infectedCount t dt 
   17     = embed (sir populationSize infectedCount) ((), steps)
   18   where
   19     n     = t / dt
   20     steps = replicate (floor n) (dt, Nothing)
   21 
   22 sir :: Double 
   23     -> Double
   24     -> SF () (Double, Double, Double)
   25 sir populationSize infectedCount 
   26     = loopPre (initSus, initInf, initRec) sir'
   27   where
   28     sir' :: SF 
   29             ((), (Double, Double, Double))
   30             ((Double, Double, Double), (Double, Double, Double))
   31     sir' = proc (_, (s, i, _r)) -> do
   32       let infectionRate  = (i * contactRate * s * infectivity) / populationSize
   33       let recoveryRate  = i / illnessDuration
   34 
   35       s' <- (initSus+) ^<< integral -< (-infectionRate)
   36       i' <- (initInf+) ^<< integral -< (infectionRate - recoveryRate)
   37       r' <- (initRec+) ^<< integral -< recoveryRate
   38 
   39       returnA -< dupe (s', i', r')
   40 
   41     dupe :: a -> (a, a)
   42     dupe a = (a, a)
   43 
   44     initSus = populationSize - infectedCount
   45     initInf = infectedCount
   46     initRec = 0
